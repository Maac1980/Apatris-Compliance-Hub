import React, { useState, useEffect } from 'react';
import { Mail, Lock, Eye, EyeOff, LogIn, ShieldAlert, MapPin, Cpu, Layers, Disc, Hammer, Wrench, Sparkles, MessageSquare, X, Send, Loader2 } from 'lucide-react';

export default function App() {

  const [password, setPassword] = useState('');

  const [isLoading, setIsLoading] = useState(false);

  // --- Gemini AI State ---
  const [aiTip, setAiTip] = useState('');
  const [isGeneratingTip, setIsGeneratingTip] = useState(false);
  const [isChatOpen, setIsChatOpen] = useState(false);
  const [chatInput, setChatInput] = useState('');
  const [chatLog, setChatLog] = useState([{ role: 'ai', text: 'Apatris AI Support online. State your query regarding welding protocols, alloy specifications, or system access.' }]);
  const [isChatting, setIsChatting] = useState(false);

  // --- Session Persistence (Check on Load) ---
  useEffect(() => {
    const savedToken = localStorage.getItem('apatris_session');
    if (savedToken) {
      console.log('Active session found. Bypassing login...');
      // Uncomment the line below when you have your routing set up
      // window.location.href = '/dashboard'; 
    }
  }, []);

  // --- Login Handler ---
  const handleLogin = (e) => {
    e.preventDefault();
    setIsLoading(true);
    
    // Simulated authentication process
    setTimeout(() => {
      setIsLoading(false);
      // Save a token to localStorage so the device remembers the user for 30 days
      localStorage.setItem('apatris_session', 'simulated_jwt_token_12345');
      console.log('Credentials accepted. Session saved.');
      
      // Redirect to the dashboard
      // window.location.href = '/dashboard';
    }, 1500);
  };

  // --- Gemini API Helpers ---
  const callGemini = async (prompt) => {
    const apiKey = ""; // API key is populated by the execution environment
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
    const payload = { contents: [{ parts: [{ text: prompt }] }] };

    let delay = 1000;
    for (let i = 0; i < 5; i++) {
      try {
        const res = await fetch(url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload)
        });
        if (!res.ok) throw new Error('API Error');
        const data = await res.json();
        return data.candidates?.[0]?.content?.parts?.[0]?.text || "No response from mainframe.";
      } catch (e) {
        if (i === 4) return "Error: Connection to Apatris AI severed. Please check local network.";
        await new Promise(r => setTimeout(r, delay));
        delay *= 2;
      }
    }
  };

  const handleGenerateTip = async () => {
    setIsGeneratingTip(true);
    const prompt = "You are the Apatris AI. Generate a single, highly technical, 2-sentence shift briefing for robotic welding operators using S600 wire. Focus on safety or efficiency. Use a gritty, professional, industrial tone. Do not use asterisks or markdown.";
    const tip = await callGemini(prompt);
    setAiTip(tip);
    setIsGeneratingTip(false);
  };

  const handleChatSubmit = async (e) => {
    e.preventDefault();
    if (!chatInput.trim() || isChatting) return;
    
    const newLog = [...chatLog, { role: 'user', text: chatInput }];
    setChatLog(newLog);
    setChatInput('');
    setIsChatting(true);

    const conversation = newLog.map(msg => `${msg.role === 'user' ? 'Operator' : 'AI'}: ${msg.text}`).join('\n');
    const prompt = `You are the Apatris AI Support System, assisting an operator at a heavy-duty welding terminal. Keep responses technical, robotic, and concise. No markdown formatting.\n\n${conversation}\nAI:`;
    
    const response = await callGemini(prompt);
    setChatLog([...newLog, { role: 'ai', text: response }]);
    setIsChatting(false);
  };

    backgroundPosition: '0 0, 0 0, 0 0, 8px 8px'

        {/* Card background */}
        <div className="bg-[#111111]/90 backdrop-blur-xl border border-stone-700/60 rounded-xl shadow-[0_20px_50px_rgba(0,0,0,0.9),inset_0_1px_0_rgba(255,255,255,0.05)] overflow-hidden relative flex flex-col md:flex-row">
          
          {/* Rivets */}
          <div className="absolute top-3 left-3 w-3 h-3 rounded-full bg-stone-500 border border-stone-800 shadow-[inset_1px_1px_2px_rgba(255,255,255,0.4),0_1px_3px_rgba(0,0,0,0.8)] z-20 hidden md:block"></div>
          <div className="absolute top-3 right-3 w-3 h-3 rounded-full bg-stone-500 border border-stone-800 shadow-[inset_1px_1px_2px_rgba(255,255,255,0.4),0_1px_3px_rgba(0,0,0,0.8)] z-20"></div>
          <div className="absolute bottom-3 left-3 w-3 h-3 rounded-full bg-stone-500 border border-stone-800 shadow-[inset_1px_1px_2px_rgba(255,255,255,0.4),0_1px_3px_rgba(0,0,0,0.8)] z-20 hidden md:block"></div>
          <div className="absolute bottom-3 right-3 w-3 h-3 rounded-full bg-stone-500 border border-stone-800 shadow-[inset_1px_1px_2px_rgba(255,255,255,0.4),0_1px_3px_rgba(0,0,0,0.8)] z-20"></div>

          {/* Top accent line */}
          <div className="absolute top-0 left-0 right-0 h-1.5 bg-gradient-to-r from-red-900 via-red-600 to-red-900 border-b border-black/80 z-20"></div>

          {/* LEFT PANEL */}
          <div className="w-full md:w-5/12 p-8 pt-12 border-b md:border-b-0 md:border-r border-stone-800/80 bg-[#161616] flex flex-col justify-between relative shadow-[inset_-10px_0_20px_rgba(0,0,0,0.4)]">
            
            <div>

                <div className="w-16 h-16 shrink-0 rounded-full border-[3px] border-stone-950 flex items-center justify-center bg-gradient-to-br from-stone-300 to-stone-500 shadow-[inset_0_2px_4px_rgba(255,255,255,0.5),0_4px_10px_rgba(0,0,0,0.8)]">
                  <span className="text-red-700 font-black text-4xl tracking-tighter drop-shadow-md" style={{ fontFamily: 'Impact, sans-serif' }}>A</span>

                  <p className="text-xs text-red-600 font-bold tracking-[0.2em] uppercase mt-1.5">Specialist Welding</p>
                </div>
              </div>

              <div className="space-y-6 mt-10">
                <div className="flex
